    4. **Create backend/app/services/skill/storage.py** implementing:
       ```python
       from minio import Minio
       from app.services.skill.errors import SkillStorageError

       class SkillStorage:
           """Manages skill package storage in MinIO (per D-30)."""

           def __init__(self, client: Minio, bucket: str = "skill-packages") -> None:
               self._client = client
               self._bucket = bucket
               self._ensure_bucket()

           def _ensure_bucket(self) -> None:
               if not self._client.bucket_exists(self._bucket):
                   self._client.make_bucket(self._bucket)

           def store_package(self, name: str, version: str, data: bytes) -> str:
               """Store ZIP package. Returns the MinIO object key."""
               key = f"{name}/{version}.zip"
               from io import BytesIO
               self._client.put_object(
                   self._bucket, key, BytesIO(data), len(data),
                   content_type="application/zip",
               )
               return key

           def get_package(self, name: str, version: str) -> bytes:
               """Retrieve ZIP package bytes from MinIO."""
               key = f"{name}/{version}.zip"
               response = self._client.get_object(self._bucket, key)
               try:
                   return response.read()
               finally:
                   response.close()
                   response.release_conn()

           def delete_package(self, name: str, version: str) -> None:
               """Remove ZIP package from MinIO."""
               key = f"{name}/{version}.zip"
               self._client.remove_object(self._bucket, key)
       ```
